    SELECT
        gal.objID, gal.u, gal.g,
        gal.r, gal.i, gal.z, s.z,
        x.plate, x.mjd, s.fiberID
    FROM galaxy gal
    JOIN specObj s on s.bestObjID = gal.objID
    JOIN plateX x on x.plateID = s.plateID
    WHERE
        s.class='galaxy' 
        AND gal.g BETWEEN 17 and 17.4 
        AND s.z < 0.05
